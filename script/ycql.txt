CREATE KEYSPACE CS5424
  WITH REPLICATION = {
    'class' : 'SimpleStrategy',
    'replication_factor' : 1
  }
AND DURABLE_WRITES = false; 

USE CS5424;

CREATE TABLE CS5424.Item (
  I_ID int,
  I_NAME text,
  I_PRICE double,
  I_IM_ID int,
  I_DATA text,
  PRIMARY KEY (I_ID)
);

CREATE TABLE CS5424.Order (
  O_W_ID int,
  O_D_ID int,
  O_ID int,
  O_C_ID int,
  O_CARRIER_ID int,
  O_OL_CNT int,
  O_ALL_LOCAL int,
  O_ENTRY_D timestamp,
  PRIMARY KEY ((O_ID, O_W_ID, O_D_ID), O_ENTRY_D)
)
WITH CLUSTERING ORDER BY (O_ENTRY_D ASC);

CREATE TABLE CS5424.OrderLine (
  OL_W_ID int,
  OL_D_ID int,
  OL_O_ID int,
  OL_NUMBER int,
  OL_I_ID int,
  OL_DELIVERY_D timestamp,
  OL_AMOUNT double,
  OL_SUPPLY_W_ID int,
  OL_QUANTITY int,
  OL_DIST_INFO text,
  PRIMARY KEY ((OL_W_ID, OL_D_ID, OL_O_ID, OL_NUMBER), OL_SUPPLY_W_ID, OL_DELIVERY_D)
)
WITH CLUSTERING ORDER BY (OL_SUPPLY_W_ID ASC, OL_DELIVERY_D ASC);

CREATE TABLE CS5424.Warehouse (
  W_ID int,
  W_NAME text,
  W_STREET_1 text,
  W_STREET_2 text,
  W_CITY text,
  W_STATE text,
  W_ZIP text,
  W_TAX double,
  W_YTD double,
  PRIMARY KEY (W_ID)
);

CREATE TABLE CS5424.District (
  D_W_ID int,
  D_ID int,
  D_NAME text,
  D_STREET_1 text,
  D_STREET_2 text,
  D_CITY text,
  D_STATE text,
  D_ZIP text,
  D_TAX double,
  D_YTD double,
  D_NEXT_O_ID int,
  PRIMARY KEY (D_W_ID, D_ID)
);

CREATE TABLE CS5424.Customer (
  C_W_ID int,
  C_D_ID int,
  C_ID int,
  C_FIRST text,
  C_MIDDLE text,
  C_LAST text,
  C_STREET_1 text,
  C_STREET_2 text,
  C_CITY text,
  C_STATE text,
  C_ZIP text,
  C_PHONE text,
  C_SINCE timestamp,
  C_CREDIT text,
  C_CREDIT_LIM double,
  C_DISCOUNT double,
  C_BALANCE double,
  C_YTD_PAYMENT double,
  C_PAYMENT_CNT int,
  C_DELIVERY_CNT int,
  C_DATA text,
  PRIMARY KEY (C_W_ID, C_D_ID, C_ID)
);

CREATE TABLE CS5424.Stock (
  S_W_ID int,
  S_I_ID int,
  S_QUANTITY int,
  S_YTD double,
  S_ORDER_CNT int,
  S_REMOTE_CNT int,
  S_DIST_01 text,
  S_DIST_02 text,
  S_DIST_03 text,
  S_DIST_04 text,
  S_DIST_05 text,
  S_DIST_06 text,
  S_DIST_07 text,
  S_DIST_08 text,
  S_DIST_09 text,
  S_DIST_10 text,
  S_DATA text,
  PRIMARY KEY (S_W_ID, S_I_ID)
);

CREATE TABLE CS5424.OrderByCustomer (
  C_W_ID int,
  C_D_ID int,
  C_ID int,
  O_ENTRY_D timestamp,
  C_FIRST text,
  C_MIDDLE text,
  C_LAST text,
  C_BALANCE double,
  C_LAST_O_ID int,
  O_CARRIER_ID int,
  PRIMARY KEY ((C_W_ID, C_D_ID, C_ID), O_ENTRY_D)
)
WITH CLUSTERING ORDER By (O_ENTRY_D DESC);

CREATE TABLE CS5424.StockByOrderLine (
  W_ID int,
  D_ID int,
  O_ID int,
  O_ENTRY_D timestamp,
  S_QUANTITY_MAP map<int, int>,
  PRIMARY KEY ((W_ID, D_ID, O_ID), O_ENTRY_D)
)
WITH CLUSTERING ORDER By (O_ENTRY_D DESC);

CREATE TABLE CS5424.OrderLineQuantityByOrder (
  W_ID int,
  D_ID int,
  O_ID int,
  O_ENTRY_D timestamp,
  OL_QUANTITY_MAP map<int, int>,
  C_FIRST text,
  C_MIDDLE text,
  C_LAST text,
  PRIMARY KEY ((W_ID, D_ID, O_ID), O_ENTRY_D)
)
WITH CLUSTERING ORDER By (O_ENTRY_D DESC);

CREATE TABLE CS5424.CustomerBalance (
  C_W_ID int,
  C_D_ID int,
  C_ID int,
  C_BALANCE double,
  C_FIRST text,
  C_MIDDLE text,
  C_LAST text,
  W_NAME text,
  D_NAME text,
  PRIMARY KEY (C_W_ID, C_D_ID, C_ID)
);
